// ---- assignment storage using localStorage ---- function assignmentKey(business, code){ return `loc:${business}||${code}`; } async function getAssignment(business, code){ try{ const data = localStorage.getItem(assignmentKey(business, code)); return data ? JSON.parse(data) : null; }catch(e){ console.error("Load Error:", e); return null; } } async function setAssignment(business, code, location){ const payload = { location, updatedAt: new Date().toISOString() }; try{ localStorage.setItem( assignmentKey(business, code), JSON.stringify(payload) ); }catch(e){ console.error("Save Error:", e); } return payload; } async function clearAssignment(business, code){ try{ localStorage.removeItem(assignmentKey(business, code)); }catch(e){ console.error("Delete Error:", e); } }
